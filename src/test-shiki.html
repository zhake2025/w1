Python 示例：

```python
def fibonacci(n):
    """计算斐波那契数列的前n项"""
    fib_sequence = []
    a, b = 0, 1  # 初始化前两个斐波那契数
    
    for _ in range(n):
        fib_sequence.append(a)
        a, b = b, a + b  # 更新a和b的值
    
    return fib_sequence

# 示例用法
n = 10  # 你可以修改这个值来计算不同数量的项
result = fibonacci(n)
print(f"斐波那契数列的前{n}项是: {result}")
```

JavaScript 示例：

```javascript
// 异步函数示例
async function fetchUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        const userData = await response.json();
        
        return {
            id: userData.id,
            name: userData.name,
            email: userData.email,
            isActive: userData.status === 'active'
        };
    } catch (error) {
        console.error('获取用户数据失败:', error);
        throw new Error('用户数据获取失败');
    }
}

// 使用示例
fetchUserData(123)
    .then(user => console.log('用户信息:', user))
    .catch(err => console.error('错误:', err));
```
